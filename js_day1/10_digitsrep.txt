
function countRepeatingDigits(N)
{
	var res = 0;
	var cnt =Array(10).fill(0);

	while (N > 0) {
		var rem = N % 10;
		cnt[rem]++;
		N = Math.floor(N / 10);
	}

	for (var i = 0; i < 10; i++) {

		if (cnt[i] > 1) {

			res++;
		}
	}
	return res;
}
var N=7312180905;

console.log(countRepeatingDigits(N));
